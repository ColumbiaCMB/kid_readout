[docs]    def sendget(self,msg,interchar_delay=0.1):
        if msg[-1] != '\n':
            msg = msg + '\n'
        for char in msg:
            self.ser.write(char)
            time.sleep(interchar_delay)
        time.sleep(1)
        resp = self.ser.read(self.ser.inWaiting())
        return resp
