[docs]def fractional_freq_to_power_guess(x,y):
    mid = 10**np.log10(x).mean()
    scale = y.max()
    params = lmfit.Parameters()
    params.add('break_point',1/mid,min=x.min(),max=x.max())
    params.add('scale',scale,min=0,max=1)
    return params
