[docs]    def do_scan(self,freqs,level=-50.0,nsamp=2048):
        self.set_level_dbm(level)
        self.hittite.on()
        data = []
        for freq in freqs:
            data.append(self.read_freq(freq,nsamp=nsamp))
        self.hittite.off()
        return freqs,np.array(data)
